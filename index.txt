    <script>
        const date_today = new Date().toISOString().split('T')[0];
        document.getElementById('fecha-reserva').min = date_today;
        document.getElementById('fecha-reserva').value = date_today;
        
        let canchaSeleccionada = null;
        let horarioSeleccionado = null;
        let fechaSeleccionada = null;
        
        function seleccionarCancha(canchaId, canchaNombre) {
            canchaSeleccionada = canchaId;
            document.getElementById('nombre-cancha').textContent = canchaNombre;
            document.getElementById('cancha-id').value = canchaId;
            document.getElementById('paso-1').classList.add('hidden');
            document.getElementById('paso-2').classList.remove('hidden');
            cargarHorarios();
        }
        
        function volverAPaso1() {
            document.getElementById('paso-2').classList.add('hidden');
            document.getElementById('paso-1').classList.remove('hidden');
        }
        
        function volverAPaso2() {
            document.getElementById('paso-3').classList.add('hidden');
            document.getElementById('paso-2').classList.remove('hidden');
        }
        
        async function cargarHorarios() {
            const fecha = document.getElementById('fecha-reserva').value;
            if (!fecha || !canchaSeleccionada) return;
            
            const response = await fetch(`/disponibilidad?cancha_id=${canchaSeleccionada}&fecha=${fecha}`);
            const data = await response.json();
            
            const container = document.getElementById('horarios-container');
            container.innerHTML = '<h3>Horarios Disponibles:</h3>';
            
            data.horarios_disponibles.forEach(horario => {
                const button = document.createElement('button');
                button.className = 'horario-btn disponible';
                button.textContent = horario;
                button.onclick = () => seleccionarHorario(horario, fecha);
                container.appendChild(button);
            });
            
            if (data.horarios_ocupados.length > 0) {
                const ocupadosDiv = document.createElement('div');
                ocupadosDiv.innerHTML = '<h3>Horarios Ocupados:</h3>';
                data.horarios_ocupados.forEach(horario => {
                    const span = document.createElement('span');
                    span.className = 'horario-btn ocupado';
                    span.textContent = horario;
                    span.style.margin = '5px';
                    ocupadosDiv.appendChild(span);
                });
                container.appendChild(ocupadosDiv);
            }
        }
        
        function seleccionarHorario(horario, fecha) {
            horarioSeleccionado = horario;
            fechaSeleccionada = fecha;
            document.getElementById('horario-seleccionado').value = horario;
            document.getElementById('fecha-seleccionada').value = fecha;
            document.getElementById('paso-2').classList.add('hidden');
            document.getElementById('paso-3').classList.remove('hidden');
        }
        
        document.getElementById('formulario-reserva').addEventListener('submit', async function(e) {
            e.preventDefault();
            
            const formData = new FormData(this);
            const datos = Object.fromEntries(formData);
            
            try {
                const response = await fetch('/reservar', {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify(datos)
                });
                
                const resultado = await response.json();
                
                if (resultado.success) {
                    document.getElementById('paso-3').classList.add('hidden');
                    document.getElementById('resultado').innerHTML = `
                        <div class="success">
                            <h3>‚úÖ ${resultado.mensaje}</h3>
                            <p><strong>Cliente:</strong> ${resultado.detalles.cliente}</p>
                            <p><strong>Cancha:</strong> ${resultado.detalles.cancha} (${resultado.detalles.tipo})</p>
                            <p><strong>Fecha:</strong> ${resultado.detalles.fecha}</p>
                            <p><strong>Horario:</strong> ${resultado.detalles.horario}</p>
                            <p><strong>M√©todo de Pago:</strong> ${resultado.detalles.metodo_pago}</p>
                            <p><strong>Monto Total:</strong> ${resultado.detalles.monto}</p>
                            <p><a href="/confirmacion/${resultado.reserva_id}">Ver confirmaci√≥n detallada</a></p>
                        </div>
                    `;
                } else {
                    document.getElementById('resultado').innerHTML = `
                        <div class="error">
                            <h3>‚ùå Error</h3>
                            <p>${resultado.error}</p>
                        </div>
                    `;
                }
            } catch (error) {
                document.getElementById('resultado').innerHTML = `
                    <div class="error">
                        <h3>‚ùå Error de conexi√≥n</h3>
                        <p>Int√©ntalo de nuevo m√°s tarde</p>
                    </div>
                `;
            }
        });
